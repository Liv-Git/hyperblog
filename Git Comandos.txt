git init = iniciar git en una carpeta

gitk = abrir en un software la historia de ramas supervisual

checkout = te permite cambiar de rama

push = llevar los datos del pc al repositorio remoto

pull = llevar los datos del repositorio a mi pc

branch = ver/cambiar de rama

cat = Me muestra la informacion del archivo en consola

git remote -v = saber la url esta vinculada a mi repositorio local

git log --all --graph --decorate --oneline = ver las ramas de forma mas gráfica y sus cambios

git tag -a v0.1 -m "descripción del tag" (serial del tag) = para crear nuevos tags

git show-branch --all = cuales son las ramas que existen y cual es su historia

git reset --merge = para cancelar un merge
